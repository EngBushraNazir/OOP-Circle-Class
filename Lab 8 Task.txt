#include<iostream>
#include<stdlib.h>
using namespace std;

class person {
private:
	char name[50];
	int age;
public:

	person() {
		strcpy_s(name, "");
		age = 0;
	}
	person(char a[50],int b) {
		strcpy_s(name, a);
		age = b;
	}
	void get_data() {
		cout << "Name : ";
		cin.get(name, 50);
		cout << "age : ";
		cin >> age;
		cin.ignore(1, '\n');

	}
	void show_data() {
		cout << "Name          : " << name << endl
			<< "age           : " << age << endl;
	}
};




class student :public person {

		person::show_data();
